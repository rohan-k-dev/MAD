package com.example.ccheckbox

import android.os.Bundle
import android.view.LayoutInflater
import android.widget.*
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val btn = findViewById<Button>(R.id.button1)

        btn.setOnClickListener {

            val ckb3 = findViewById<CheckBox>(R.id.checkBox3)
            val ckb4 = findViewById<CheckBox>(R.id.checkBox4)

            val sb = StringBuilder()

            if (ckb3.isChecked)
                sb.append(ckb3.text).append(" ")

            if (ckb4.isChecked)
                sb.append(ckb4.text)

            // Inflate custom toast layout
            val inflater = LayoutInflater.from(this)
            val layout = inflater.inflate(R.layout.custom_toast, null)

            val text = layout.findViewById<TextView>(R.id.toast_text)
            text.text = if (sb.isNotEmpty()) sb.toString() else "Nothing Selected"

            val toast = Toast(this)
            toast.duration = Toast.LENGTH_LONG
            toast.view = layout
            toast.show()
        }
    }
}
