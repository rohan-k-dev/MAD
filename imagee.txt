


instructions
1. Add images in:
   app > res > drawable

Image names:
donut.jpg
chocolate.jpg
icecream.jpg

2. Add strings in:
   app > res > values > strings.xml

3. Add XML code in:
   app > res > layout > activity_main.xml

4. Add Kotlin code in:
   app > kotlin+java > package name > MainActivity.kt

strings.xml--->

<resources>
    <string name="app_name">imageview</string>
    <string name="donut">donuts are very sweet and yummy</string>
    <string name="choco">chocolates are so tasty</string>
    <string name="icecream">icecream are so cold</string>
</resources>





package com.example.imageview

import android.support.v7.app.AppCompatActivity
import android.os.Bundle
import android.view.View
import android.widget.Toast

class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

    }
    fun donut(view: View) {
        Toast.makeText(this,"donut clicked",Toast.LENGTH_SHORT).show()
    }
    fun choco(view: View) {
        Toast.makeText(this,"chocolate clicked",Toast.LENGTH_SHORT).show()
    }
    fun icecreamm(view: View) {
        Toast.makeText(this,"icecreamm clicked",Toast.LENGTH_SHORT).show()
    }
}
